from django import template

register = template.Library()

@register.assignment_tag
def robot_count():
	robot_counts = dict()

	robot_counts["line_follower"] = 0
	robot_counts["line_follower_junior"] = 0
	robot_counts["drone"] = 0
	robot_counts["micro_sumo"] = 0
	robot_counts["construction"] = 0
	robot_counts["scenario"] = 0
	robot_counts["innovative"] = 0
	robot_counts["color_selecting"] = 0

	return robot_counts
